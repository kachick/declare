def report
  unless @failures.empty?
    report_detail
    puts '-' * 78
  end

  failure_count = failure_count()
  puts "#{@scope_summaries.length} scopes, #{@declare_counter} assertions, #{failure_count} failures"

  exit(failure_count)
end
